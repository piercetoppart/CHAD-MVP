// Confidence Scoring Engine
class ConfidenceEngine {
  private readonly DECAY_RATE = 0.95; // Daily confidence decay without wins
  private readonly BOOST_RATE = 1.15; // Win boost multiplier
  
  calculateConfidenceScore(user: User): ConfidenceScore {
    const components = {
      recentPerformance: this.assessRecentPerformance(user),
      skillAlignment: this.calculateSkillConfidence(user),
      marketPosition: this.assessMarketPosition(user),
      socialValidation: this.calculateSocialValidation(user),
      momentum: this.calculateMomentum(user)
    };
    
    // Apply psychological weighting
    const weights = this.getPersonalizedWeights(user.personalityType);
    
    const rawScore = Object.entries(components).reduce((total, [key, value]) => {
      return total + (value.score * weights[key]);
    }, 0);
    
    // Apply time-based modifiers
    const timeAdjusted = this.applyTimeDecay(rawScore, user.lastWin);
    
    // Ensure score bounds and smoothing
    const smoothed = this.smoothConfidenceTransition(
      user.previousConfidence,
      timeAdjusted
    );
    
    return {
      score: Math.max(20, Math.min(95, smoothed)), // Never below 20 or above 95
      components,
      trend: this.calculateTrend(user.confidenceHistory),
      interventions: this.suggestInterventions(smoothed, components)
    };
  }
  
  assessRecentPerformance(user: User): ComponentScore {
    const recentWins = user.wins.filter(w => 
      w.date > Date.now() - 7 * 24 * 60 * 60 * 1000
    );
    
    const recentLosses = user.rejections.filter(r => 
      r.date > Date.now() - 7 * 24 * 60 * 60 * 1000
    );
    
    // Psychological principle: Wins count more than losses
    const winImpact = recentWins.length * 10;
    const lossImpact = recentLosses.length * 3; // Losses hurt less
    
    const performanceScore = Math.max(0, 50 + winImpact - lossImpact);
    
    return {
      score: performanceScore,
      factors: {
        wins: recentWins.length,
        losses: recentLosses.length,
        ratio: recentWins.length / (recentLosses.length || 1)
      },
      insight: this.generatePerformanceInsight(performanceScore)
    };
  }
  
  // Personalized weight system based on user psychology
  getPersonalizedWeights(personalityType: PersonalityType): WeightMap {
    const baseWeights = {
      recentPerformance: 0.30,
      skillAlignment: 0.25,
      marketPosition: 0.20,
      socialValidation: 0.15,
      momentum: 0.10
    };
    
    // Adjust based on personality
    switch (personalityType) {
      case 'analytical':
        return {
          ...baseWeights,
          marketPosition: 0.30, // Data-driven users value market position
          socialValidation: 0.10
        };
        
      case 'creative':
        return {
          ...baseWeights,
          skillAlignment: 0.35, // Creatives value skill expression
          marketPosition: 0.15
        };
        
      case 'relationship-focused':
        return {
          ...baseWeights,
          socialValidation: 0.25, // Social proof matters more
          recentPerformance: 0.25
        };
        
      default:
        return baseWeights;
    }
  }
  
  // Smooth confidence transitions to prevent jarring changes
  smoothConfidenceTransition(previous: number, target: number): number {
    const MAX_DAILY_CHANGE = 15; // Maximum 15-point swing per day
    const difference = target - previous;
    
    if (Math.abs(difference) <= MAX_DAILY_CHANGE) {
      return target;
    }
    
    // Gradual transition
    return previous + (Math.sign(difference) * MAX_DAILY_CHANGE);
  }
  
  // Generate actionable interventions based on confidence level
  suggestInterventions(score: number, components: ComponentMap): Intervention[] {
    const interventions: Intervention[] = [];
    
    if (score < 40) {
      // Low confidence interventions
      interventions.push({

        expectedImpact: '+5 confidence points'
      });

        priority: 'high',
        action: 'Apply for premium opportunity',
        reason: 'Your confidence supports bigger goals',
        expectedImpact: 'Potential 2x normal rate'
      });
    }
    
    // Component-specific interventions
    Object.entries(components).forEach(([component, data]) => {
      if (data.score < 40) {
        interventions.push(this.getComponentIntervention(component, data));
      }
    });
    
    return interventions.sort((a, b) => 
      this.getPriorityWeight(b.priority) - this.getPriorityWeight(a.priority)
    );
  }

// Smart Notification System with Psychological Awareness
class PsychologicalNotificationEngine {
  private readonly MAX_DAILY_NOTIFICATIONS = 3;
  private readonly QUIET_HOURS = { start: 21, end: 8 };
  private notificationHistory: NotificationLog[] = [];
  
  async shouldSendNotification(
    notification: PotentialNotification,
    user: User
  ): Promise {
    // Multi-factor decision making
    const factors = await this.evaluateFactors({
      user,
      notification,
      currentTime: new Date(),
      userState: await this.assessUserState(user),
      recentActivity: await this.getRecentActivity(user)
    });
    
    // Decision tree
    if (!this.isAppropriateTime(factors.currentTime, user)) {
      return { send: false, reason: 'quiet_hours', reschedule: true };
    }
    
    if (this.hasReachedDailyLimit(user)) {
      return { send: false, reason: 'daily_limit', queue: true };
    }
    
    if (factors.userState.stress > 70) {
      return this.handleHighStressUser(notification, factors);
    }
    
    if (factors.userState.engagement < 30) {
      return this.handleDisengagedUser(notification, factors);
    }
    
    // Calculate notification value
    const value = this.calculateNotificationValue(notification, factors);
    
    if (value.score > 70) {
      return {
        send: true,
        channel: this.selectOptimalChannel(user, notification),
        timing: this.optimizeTiming(factors),
        content: this.personalizeContent(notification, factors)
      };
    }
    
    return { send: false, reason: 'low_value', reconsiderIn: '2_hours' };
  }
  
  // Stress-aware notification handling
  handleHighStressUser(
    notification: PotentialNotification,
    factors: EvaluationFactors
  ): NotificationDecision {
    // Only send if it reduces stress
    const stressReducers = [
      'quick_win_available',
      'positive_feedback',
      'streak_celebration',
      'support_message'
    ];
    
    if (stressReducers.includes(notification.type)) {
      return {
        send: true,
        channel: 'gentle_push', // Softer notification style
        content: this.createCalmingMessage(notification),
        priority: 'low' // Don't add urgency
      };
    }
    
    // Delay stressful notifications
    if (notification.type.includes('deadline') || notification.type.includes('urgent')) {
      return {
        send: false,
        reason: 'user_stressed',
        reschedule: {
          when: 'tomorrow_morning',
          condition: 'stress_below_50'
        }
      };
    }
    
    return { send: false, reason: 'protecting_wellbeing' };
  }
  
  // Re-engagement strategy for inactive users
  handleDisengagedUser(
    notification: PotentialNotification,
    factors: EvaluationFactors
  ): NotificationDecision {
    const reengagementTypes = [
      'major_opportunity',
      'peer_success_story',
      'easy_win',
      'personal_milestone'
    ];
    
    if (!reengagementTypes.includes(notification.type)) {
      // Regular notifications won't work
      return { send: false, reason: 'user_disengaged' };
    }
    
    // Craft re-engagement message
    const lastSuccess = factors.user.lastWin;
    const daysSinceSuccess = (Date.now() - lastSuccess.date) / (1000 * 60 * 60 * 24);
    
    let message: NotificationContent;
    
    if (daysSinceSuccess < 7) {
      message = {
        title: 'Building on your recent win! üéâ',
        body: `Your ${lastSuccess.project} success opened new doors. Check out this similar opportunity.`,
        emotion: 'encouraging',
        action: 'view_opportunity'
      };
    } else {
      message = {
        title: 'Perfect opportunity for you',
        body: 'This one matches your best work. Low competition, high success chance.',
        emotion: 'supportive',
        action: 'quick_view'
      };
    }
    
    return {
      send: true,
      channel: 'email', // Less intrusive for disengaged users
      content: message,
      followUp: {
        if: 'no_action',
        after: '48_hours',
        message: 'simplified_opportunity'
      }
    };
  }
  
  // Personalization engine
  personalizeContent(
    notification: PotentialNotification,
    factors: EvaluationFactors
  ): NotificationContent {
    const templates = this.loadTemplates(notification.type);
    const user = factors.user;
    
    // Personality-based language
    const languageStyle = this.getLanguageStyle(user.personalityType);
    
    // Time-based energy adjustment
    const energyLevel = this.predictEnergyLevel(factors.currentTime, user);
    
    // Recent context incorporation
    const recentContext = this.extractRecentContext(factors.recentActivity);
    
    // Build personalized message
    const personalizedTemplate = templates[languageStyle];
    
    const variables = {
      userName: user.firstName,
      timeOfDay: this.getTimeGreeting(factors.currentTime),
      recentWin: recentContext.lastSuccess,
      currentStreak: user.currentStreak,
      ...notification.data
    };
    
    const content = this.interpolateTemplate(personalizedTemplate, variables);
    
    // Adjust for energy level
    if (energyLevel < 40) {
      content.body = this.simplifyMessage(content.body);
      content.action = this.simplifyAction(content.action);
    }
    
    return content;
  }
  
  // Channel selection based on urgency and user preference
  selectOptimalChannel(user: User, notification: PotentialNotification): Channel {
    const preferences = user.notificationPreferences;
    const urgency = notification.urgency || 'normal';
    
    // Channel effectiveness matrix
    const effectiveness = {
      push: {
        urgent: 0.9,
        normal: 0.7,
        low: 0.3
      },
      email: {
        urgent: 0.6,
        normal: 0.8,
        low: 0.9
      },
      in_app: {
        urgent: 0.5,
        normal: 0.9,
        low: 0.8
      }
    };
    
    // User preference weights
    const preferenceWeights = {
      push: preferences.push ? 1.0 : 0,
      email: preferences.email ? 1.0 : 0,
      in_app: preferences.inApp ? 1.0 : 0
    };
    
    // Calculate optimal channel
    let bestChannel = 'in_app';
    let bestScore = 0;
    
    Object.entries(effectiveness).forEach(([channel, scores]) => {
      const score = scores[urgency] * preferenceWeights[channel];
      if (score > bestScore) {
        bestScore = score;
        bestChannel = channel;
      }
    });
    
    return bestChannel as Channel;
  }
}
