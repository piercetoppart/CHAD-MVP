// Morning Anxiety Reducer System
class MorningDashboard {
  constructor(user) {
    this.user = user;
    this.timezone = user.timezone;
    this.preferences = user.morningPreferences;
  }
  
  async generateMorningView() {
    const overnight = await this.getOvernightActivity();
    const priorities = await this.calculateMorningPriorities();
    const energyLevel = this.predictEnergyLevel();
    
    return {
      greeting: this.getPersonalizedGreeting(),
      summary: this.createCalmingSummary(overnight),
      priorities: this.limitChoices(priorities, 3), // Reduce overwhelm
      gamePlan: this.generateGamePlan(energyLevel),
      motivationalContext: this.getMorningMotivation()
    };
  }
  
  createCalmingSummary(data) {
    // Instead of "You missed 47 jobs!", show:
    const summary = {
      totalNew: data.newJobs.length,
      relevant: data.newJobs.filter(j => j.matchScore > 80).length,
      urgent: data.newJobs.filter(j => j.urgency === 'high').length,
      
      // Calming framing
      message: `${data.relevant} great matches appeared. The best ones are still fresh!`,
      
      // Reduce FOMO
      context: `Average freelancer applies to 2-3 jobs. You're right on track.`,
      
      // Focus direction
      action: 'Let\'s focus on quality over quantity today.'
    };
    
    return summary;
  }
  
  predictEnergyLevel() {
    const hour = new Date().getHours();
    const dayOfWeek = new Date().getDay();
    const historicalPerformance = this.user.performanceByHour[hour];
    
    // Factors affecting morning energy
    const factors = {
      timeOptimal: hour >= 7 && hour <= 10,
      weekday: dayOfWeek >= 1 && dayOfWeek <= 5,
      historicalSuccess: historicalPerformance?.successRate > 0.3,
      recentWins: this.user.recentWins > 0
    };
    
    const energyScore = Object.values(factors).filter(Boolean).length;
    
    return {
      level: energyScore,
      maxLevel: 4,
      recommendation: this.getEnergyRecommendation(energyScore),
      optimalTasks: this.getTasksForEnergy(energyScore)
    };
  }
}

// Smart Stop System - Preventing Analysis Paralysis
class SmartStopSystem {
  constructor(userBehavior) {
    this.behavior = userBehavior;
    this.thresholds = {
      optimalBrowsing: 10,
      warningPoint: 15,
      hardStop: 20
    };
    this.sessionData = this.initSession();
  }
  
  trackJobView(jobData) {
    this.sessionData.jobsViewed.push({
      job: jobData,
      timestamp: Date.now(),
      timeSpent: 0,
      actions: []
    });
    
    const viewCount = this.sessionData.jobsViewed.length;
    
    // Progressive interventions
    if (viewCount === this.thresholds.optimalBrowsing) {
      return this.generateOptimalStopIntervention();
    } else if (viewCount === this.thresholds.warningPoint) {
      return this.generateWarningIntervention();
    } else if (viewCount >= this.thresholds.hardStop) {
      return this.generateHardStopIntervention();
    }
    
    return null;
  }
  
  generateOptimalStopIntervention() {
    const qualityMatches = this.sessionData.jobsViewed
      .filter(j => j.job.matchScore > 80)
      .sort((a, b) => b.job.matchScore - a.job.matchScore)
      .slice(0, 3);
    
    return {
      type: 'gentle_nudge',
      ui: 'modal',
      content: {
        headline: 'You\'ve Found Great Matches! üéØ',
        message: `${qualityMatches.length} excellent opportunities identified`,
        psychology: 'Analysis quality drops after 10 jobs',
        matches: qualityMatches,
        actions: [
          {
            label: 'Apply to Top 3',
            action: 'batch_apply',
            style: 'primary',
            psychology: 'commitment_device'
          },
          {
            label: 'Browse 5 More',
            action: 'continue_limited',
            style: 'secondary',
            psychology: 'controlled_choice'
          }
        ]
      },
      timing: {
        delay: 2000, // Show after 2 seconds on 10th job
        dismissible: true,
        autoProgress: false
      }
    };
  }
  
  analyzeScrollPattern() {
    const recentViews = this.sessionData.jobsViewed.slice(-5);
    const avgTimePerJob = recentViews.reduce((sum, view) => 
      sum + view.timeSpent, 0) / recentViews.length;
    
    return {
      pattern: avgTimePerJob < 10 ? 'skimming' : 'analyzing',
      quality: avgTimePerJob > 30 ? 'good' : 'declining',
      recommendation: this.getScrollRecommendation(avgTimePerJob)
    };
  }
  
  // Batch application mode to reduce friction
  enableBatchMode(selectedJobs) {
    return {
      template: this.generateSmartTemplate(selectedJobs),
      customizations: selectedJobs.map(job => ({
        jobId: job.id,
        clientType: job.psychologyProfile,
        keyPoints: this.extractKeyPoints(job),
        opener: this.generateOpener(job),
        closer: this.generateCloser(job)
      })),
      timeEstimate: selectedJobs.length * 3, // minutes
      successPrediction: this.batchSuccessPrediction(selectedJobs)
    };
  }
  
  // Psychology: Make stopping feel like winning, not quitting
  createStoppingRewards() {
    return {
      immediate: {
        message: 'üèÜ Focused Freelancer Badge Earned!',
        points: 50,
        benefit: 'Unlock priority job alerts'
      },
      future: {
        message: 'Come back in 2 hours for fresh opportunities',
        benefit: 'Jobs posted while you work have less competition'
      },
      social: {
        message: 'You analyzed more efficiently than 73% of users',
        comparison: 'Top freelancers apply to fewer, better jobs'
      }
    };
  }
}

// Proposal Procrastination Breaker
class ProposalConfidenceSystem {
  constructor(user) {
    this.user = user;
    this.draftStorage = new DraftStorage();
    this.confidenceFactors = this.initConfidenceFactors();
  }
  
  initConfidenceFactors() {
    return {
      // Content factors
      hasSpecificExample: { weight: 15, present: false },
      mentionsTimeline: { weight: 10, present: false },
      includesQuestion: { weight: 8, present: false },
      personalizedOpening: { weight: 12, present: false },
      clearPricing: { weight: 10, present: false },
      
      // Style factors
      professionalTone: { weight: 8, present: false },
      appropriate_length: { weight: 7, present: false },
      noTypos: { weight: 5, present: false },
      
      // Psychology factors
      clientTypeMatch: { weight: 15, present: false },
      urgencyAddressed: { weight: 10, present: false }
    };
  }
  
  analyzeProposal(proposalText, jobData) {
    const analysis = {
      currentScore: 40, // Base confidence
      factors: this.assessFactors(proposalText, jobData),
      improvements: this.generateImprovements(proposalText, jobData),
      prediction: this.predictSuccess(proposalText, jobData)
    };
    
    // Calculate total score
    Object.entries(analysis.factors).forEach(([factor, data]) => {
      if (data.present) {
        analysis.currentScore += data.weight;
      }
    });
    
    return analysis;
  }
  
  generateImprovements(proposalText, jobData) {
    const improvements = [];
    const clientType = jobData.psychologyProfile;
    
    // Check for missing elements
    if (!proposalText.match(/similar|previous|last/i)) {
      improvements.push({
        type: 'add_example',
        confidence_boost: 15,
        suggestion: `Add: "I completed a similar ${jobData.category} project last month"`,
        placement: 'after_opening',
        rationale: 'Specific examples increase trust by 40%'
      });
    }
    
    if (clientType === 'time-starved' && !proposalText.match(/days|weeks|timeline/i)) {
      improvements.push({
        type: 'add_timeline',
        confidence_boost: 12,
        suggestion: 'Add specific delivery timeline',
        example: 'I can deliver the first version in 5 days',
        rationale: 'Time-starved clients need certainty'
      });
    }
    
    // Language improvements
    const weakPhrases = {
      'I think': 'I recommend',
      'I might be able to': 'I will',
      'I\'m not sure but': '',
      'hopefully': 'will',
      'maybe': ''
    };
    
    Object.entries(weakPhrases).forEach(([weak, strong]) => {
      if (proposalText.includes(weak)) {
        improvements.push({
          type: 'strengthen_language',
          confidence_boost: 5,
          suggestion: `Replace "${weak}" with "${strong}"`,
          rationale: 'Confident language increases response rate by 23%'
        });
      }
    });
    
    return improvements.sort((a, b) => b.confidence_boost - a.confidence_boost);
  }
  
  // Auto-send feature with safeguards
  setupAutoSend(proposalId, confidenceThreshold = 80) {
    const proposal = this.draftStorage.get(proposalId);
    
    return {
      scheduled: new Date(Date.now() + 24 * 60 * 60 * 1000), // 24 hours
      conditions: {
        minConfidence: confidenceThreshold,
        maxEdits: 10, // Prevent over-editing
        timeSpent: 600 // Minimum 10 minutes of editing
      },
      safeguards: {
        finalReview: true,
        notificationBefore: '1 hour',
        cancelAnytime: true
      },
      psychologyNote: 'Deadlines reduce procrastination by 67%'
    };
  }
  
  // Reduce rejection fear through framing
  reframeRejection(proposal) {
    return {
      mindset: 'experiment',
      messages: [
        'This proposal is version 1.0 - each one gets better',
        'Average freelancer sends 10 proposals per win',
        'Rejection often means bad client fit (dodged a bullet!)',
        'Each no provides data for improvement'
      ],
      visualization: this.createSuccessVisualization(),
      backup: this.suggestBackupApplications(proposal)
    };
  }
}

// Streak Psychology Management System
class StreakManager {
  constructor(user) {
    this.user = user;
    this.streaks = this.loadStreakHistory();
    this.interventions = new InterventionEngine();
  }
  
  trackApplicationOutcome(outcome) {
    this.streaks.current.push(outcome);
    
    const analysis = this.analyzeStreak();
    const intervention = this.selectIntervention(analysis);
    
    return {
      streakType: analysis.type,
      length: analysis.length,
      intervention: intervention,
      psychologicalState: this.assessPsychologicalState(analysis)
    };
  }
  
  analyzeStreak() {
    const recent = this.streaks.current.slice(-5);
    const wins = recent.filter(o => o.type === 'win').length;
    const losses = recent.filter(o => o.type === 'loss').length;
    
    if (losses >= 3 && wins === 0) {
      return {
        type: 'losing',
        length: losses,
        severity: this.calculateSeverity(recent),
        pattern: this.identifyLossPattern(recent)
      };
    } else if (wins >= 2) {
      return {
        type: 'winning',
        length: wins,
        momentum: this.calculateMomentum(recent),
        opportunity: 'high'
      };
    }
    
    return { type: 'neutral', length: 0 };
  }
  
  // Losing streak interventions
  protectFromLosingStreak(analysis) {
    const interventions = [];
    
    // 1. Reframe the losses
    interventions.push({
      type: 'cognitive_reframe',
      content: this.generateReframe(analysis.pattern),
      priority: 'immediate'
    });
    
    // 2. Provide easy wins
    const easyWins = this.findEasyWins();
    interventions.push({
      type: 'sure_wins',
      jobs: easyWins,
      message: 'These jobs have 85%+ success rate for you',
      psychology: 'Build confidence with guaranteed wins'
    });
    
    // 3. Skill validation
    interventions.push({
      type: 'skill_reminder',
      content: {
        recentSuccesses: this.user.successHistory.slice(-5),
        earnings: this.user.totalEarnings,
        clientTestimonials: this.user.testimonials
      },
      message: 'Remember what you\'ve achieved'
    });
    
    // 4. Strategic pause option
    if (analysis.severity > 7) {
      interventions.push({
        type: 'strategic_pause',
        suggestion: 'Take a 24-hour break to reset',
        benefits: [
          'Fresh perspective tomorrow',
          'Avoid desperation applications',
          'Return with renewed energy'
        ],
        alternative: this.generateAlternativeActivities()
      });
    }
